This module contains interfaces of methods related to KeyValue_

authors: Dr. Vikas Sharma

Usage

fortran program main use easifem implicit none type( keyvalue_ ) :: obj real( dfp ) :: vec( 3 ), mat( 3, 3 ) call random_number( vec ) call random_number( mat ) obj = keyvalue( 'real-rank-0', 1.0_dfp ) obj = 1.0_dfp call display( obj, 'obj' ) obj = keyvalue( 'real-rank-1', vec ) obj = [1.0_dfp, 1.0_dfp, 1.0_dfp] call display( obj, 'obj' ) obj = keyvalue( 'real-rank-2', mat ) call display( obj, 'obj' ) end program main
